// services/nanobanana-pro.service.js
import { GoogleGenerativeAI } from '@google/generative-ai';
import fs from 'fs/promises';
import path from 'path';

// Используем APIYI как прокси-сервис — он на 44% дешевле KIE и поддерживает нативный формат Gemini [citation:8][citation:10]
const API_ENDPOINT = 'https://vip.apiyi.com';
const MODEL_NAME = 'gemini-3-pro-image-preview'; // Nano Banana Pro [citation:2][citation:10]

/**
 * Генерирует премиальную карточку товара через Nano Banana Pro (Gemini 3 Pro Image)
 * @param {Object} productData - данные от OpenAI Vision
 * @param {string} productData.name - название товара
 * @param {string} productData.description - описание
 * @param {Array} productData.features - характеристики
 * @param {string} productData.category - категория
 * @param {number} price - цена товара
 * @param {string} referenceImageBase64 - опционально, исходное фото товара (base64) для лучшего сохранения внешнего вида
 * @returns {Promise<Object>}
 */
export async function generatePremiumProductCard(productData, price, referenceImageBase64 = null) {
  try {
    const apiKey = process.env.GEMINI_API_KEY || process.env.KIE_AI_API_KEY;
    if (!apiKey) {
      throw new Error('API ключ не найден. Добавьте GEMINI_API_KEY или KIE_AI_API_KEY в Secrets');
    }

    // Настраиваем клиент с эндпоинтом APIYI (дешевле и стабильнее прямого KIE) [citation:8]
    const genAI = new GoogleGenerativeAI(apiKey);
    const model = genAI.getGenerativeModel(
      { model: MODEL_NAME },
      { apiEndpoint: API_ENDPOINT }
    );

    // Формируем детальный промпт для профессиональной карточки
    const prompt = buildProductCardPrompt(productData, price);
    
    console.log('🍌 Генерируем премиум-карточку через Nano Banana Pro...');
    console.log('Разрешение: 4K, соотношение: 1:1');

    // Готовим контент для запроса
    const contentParts = [];
    
    // Если передано референсное изображение, добавляем его для лучшего сохранения внешнего вида товара [citation:2][citation:10]
    if (referenceImageBase64) {
      // Определяем MIME тип по заголовку base64
      const mimeType = referenceImageBase64.startsWith('/9j/') ? 'image/jpeg' : 'image/png';
      const cleanBase64 = referenceImageBase64.replace(/^data:image\/\w+;base64,/, '');
      
      contentParts.push({
        inlineData: {
          mimeType: mimeType,
          data: cleanBase64
        }
      });
      console.log('📸 Референсное изображение добавлено в запрос');
    }
    
    // Добавляем текстовый промпт
    contentParts.push(prompt);

    // Генерируем изображение с параметрами качества [citation:2][citation:10]
    const generationConfig = {
      responseModalities: ['IMAGE'],      // Обязательно для Gemini Image [citation:2]
      resolution: '4K',                   // 4K — максимальное качество [citation:7]
      aspectRatio: '1:1',                // Квадрат для маркетплейсов
      temperature: 0.7,
      topP: 0.95
    };

    console.log('⏳ Отправляем запрос... (может занять 10-20 секунд)');

    const result = await model.generateContent({
      contents: [{ role: 'user', parts: contentParts }],
      generationConfig
    });

    const response = result.response;
    
    // Извлекаем изображение из ответа
    for (const part of response.candidates[0].content.parts) {
      if (part.inlineData && part.inlineData.mimeType?.startsWith('image/')) {
        const imageBuffer = Buffer.from(part.inlineData.data, 'base64');
        
        // Сохраняем во временную папку (опционально)
        const outputPath = path.join(process.cwd(), 'temp', `card_${Date.now()}.png`);
        await fs.mkdir(path.dirname(outputPath), { recursive: true });
        await fs.writeFile(outputPath, imageBuffer);
        
        console.log('✅ Карточка сгенерирована, размер:', (imageBuffer.length / 1024).toFixed(0), 'KB');
        
        return {
          success: true,
          imageBuffer,
          imagePath: outputPath,
          mimeType: part.inlineData.mimeType,
          resolution: '4K'
        };
      }
    }

    throw new Error('API не вернул изображение. Проверьте промпт на наличие запрещённого контента.');

  } catch (error) {
    console.error('❌ Nano Banana Pro error:', error);
    
    // Расшифровываем частые ошибки [citation:2][citation:3]
    let userMessage = error.message;
    if (error.message.includes('401')) userMessage = 'Неверный API-ключ. Проверьте GEMINI_API_KEY в Secrets.';
    if (error.message.includes('402')) userMessage = 'Недостаточно кредитов. Пополните баланс в KIE/APIYI.';
    if (error.message.includes('429')) userMessage = 'Превышен лимит запросов. Попробуйте позже.';
    if (error.message.includes('500')) userMessage = 'Сервис временно недоступен. Повторите через минуту.';
    
    return { 
      success: false, 
      error: userMessage,
      technical: error.message
    };
  }
}

/**
 * Формирует профессиональный промпт для карточки товара
 */
function buildProductCardPrompt(productData, price) {
  const featuresList = productData.features?.map(f => `• ${f}`).join('\n') || '';
  
  return `Создай профессиональную карточку товара для маркетплейса (Wildberries/Ozon).

## Товар
Название: ${productData.name}
Категория: ${productData.category}
Цена: ${price} руб.

## Описание
${productData.description}

## Ключевые характеристики
${featuresList}

## Требования к дизайну
1. Формат: квадратное изображение 1:1
2. Стиль: премиальная товарная съёмка, белый фон, студийное освещение
3. На изображении размести:
   - Крупный заголовок с названием товара в верхней части
   - Цену в правом нижнем углу (крупно, жирно)
   - Иконки-бейджи с 3-4 ключевыми характеристиками вокруг товара
4. Все тексты на изображении должны быть на РУССКОМ языке
5. Шрифты: современные, жирные, хорошо читаемые на мобильных устройствах
6. Товар должен выглядеть дорого и привлекательно

## Качество
- Разрешение 4K
- Реалистичная текстура материалов
- Чёткие края, без артефактов
- Профессиональный цветокоррекция

Создай изображение, которое заставит клиента нажать "Купить".`;
}

/**
 * Проверяет баланс кредитов (для APIYI или KIE)
 */
export async function checkCredits() {
  try {
    const apiKey = process.env.GEMINI_API_KEY || process.env.KIE_AI_API_KEY;
    if (!apiKey) return { success: false, error: 'API ключ не настроен' };
    
    // Для APIYI и KIE используем единый эндпоинт проверки [citation:2]
    const response = await fetch('https://api.kie.ai/api/v1/user/credits', {
      headers: { 'Authorization': `Bearer ${apiKey}` }
    });
    
    if (!response.ok) return { success: false };
    const data = await response.json();
    
    return {
      success: true,
      credits: data.credits,
      totalUsed: data.totalUsed,
      // Предупреждаем, если осталось мало кредитов
      lowCredits: data.credits < 20
    };
  } catch (error) {
    console.warn('Не удалось проверить баланс:', error.message);
    return { success: false };
  }
}
